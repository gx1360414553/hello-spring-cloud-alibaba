package com.alibaba.nacos.api.selector;

/**
 * The types of selector accepted by Nacos
 *
 * @author nkorange
 * @since 0.7.0
 */
public enum SelectorType {
    /**
     * not match any type
     */
    unknown,
    /**
     * not filter out any entity
     */
    none,
    /**
     * select by label
     */
    label
}
